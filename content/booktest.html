\[\begin{gather*} A = \begin{bmatrix} 1 & \phantom{-}1 & 1 \\ \hline 1 & -1 & 0 \\ \end{bmatrix} \\ b = \begin{bmatrix} 1 \\ 0 \\ \end{bmatrix} \end{gather*}\]
